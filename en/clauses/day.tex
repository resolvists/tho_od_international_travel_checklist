\section{Travel Day}
\label{sec:day}

\subsection{Before Leaving}

\begin{tabular}{p{9cm} p{8cm}}
$\square$ Hold mail delivery & $\square$ Arrange pet care \\ 
$\square$ Set thermostat appropriately & $\square$ Unplug appliances \\ 
$\square$ Take out trash & $\square$ Lock all windows and doors \\ 
\end{tabular}

\subsection{At the airport}

\begin{tabular}{p{9cm} p{8cm}}
$\square$ Arrive 3 hours early (international) & $\square$ Have passport and boarding pass ready \\ 
$\square$ Check duty-free restrictions & $\square$ Stay hydrated \\  
\end{tabular}
